                  </td>

              ))}

            </tbody>

          </table>

      </Card>

    </div>

const ProjectDetails = ({ project, onBack, userRole, onUpdateProject }) => {

  const [activeTab, setActiveTab] = useState('summary');

  const [isEditing, setIsEditing] = useState(false);

  const [editedProject, setEditedProject] = useState(project);


  // Sync local state if the main project prop changes from the outside

  useEffect(() => {

    setEditedProject(project);

  }, [project]);


  const handleInputChange = (e) => {

      const { name, value } = e.target;

      setEditedProject(prev => ({...prev, [name]: name === 'budget' ? parseFloat(value) || 0 : value}));

  };


  const handleNestedInputChange = (area, field, value) => {

      setEditedProject(prev => ({

          ...prev,

          [area]: {

              ...prev[area],

              [field]: value

          }

      }))

  };


  const handleSave = () => {

      onUpdateProject(editedProject);

      setIsEditing(false);

  }


  const isManagerOfProject = userRole === 'Project Manager' && (project.manager === "Alex Green" || project.manager === "Diana Prince");

  

  const tabContent = {

    summary: <SummaryTab 

        isEditing={isEditing}

        editedProject={editedProject}

        project={project}

        handleInputChange={handleInputChange}

        handleNestedInputChange={handleNestedInputChange}

    />,

    schedule: <GanttTab project={project} />,

    stakeholders: <StakeholdersTab project={project} />,

    risks: <RisksAndChangesTab project={project} />,

  };

  

  return (

    <div className="space-y-6">

       <div className="flex justify-between items-center">

        <button onClick={onBack} className="flex items-center text-blue-600 hover:underline font-semibold">

          <ArrowRight className="transform rotate-180 mr-2" size={16} />

          Back to Dashboard

        </button>

        <div className="flex items-center space-x-2">

            {isManagerOfProject && (

                isEditing ? (

                    <button onClick={handleSave} className="bg-green-500 text-white px-4 py-2 rounded-lg hover:bg-green-600 flex items-center"><Save size={16} className="mr-2"/>Save Changes</button>

                ) : (

                    <button onClick={() => setIsEditing(true)} className="bg-blue-500 text-white px-4 py-2 rounded-lg hover:bg-blue-600 flex items-center"><Edit size={16} className="mr-2"/>Edit Project</button>

                )

            )}

            {userRole === 'Higher Management' && project.approval === 'Pending' && (

              <div className="flex space-x-2">

                  <button className="bg-green-500 text-white px-4 py-2 rounded-lg hover:bg-green-600 flex items-center"><CheckCircle size={16} className="mr-2"/>Approve</button>

                  <button className="bg-red-500 text-white px-4 py-2 rounded-lg hover:bg-red-600 flex items-center"><XCircle size={16} className="mr-2"/>Reject</button>

              </div>

            )}

        </div>

      </div>

      

      <Card>

          <div className="flex flex-col sm:flex-row justify-between sm:items-center">

              <div>

                 {isEditing ? (

                     <input type="text" name="name" value={editedProject.name} onChange={handleInputChange} className="text-3xl font-bold text-gray-900 border-gray-300 rounded-md shadow-sm" />

                 ) : (

                    <h1 className="text-3xl font-bold text-gray-900">{project.name}</h1>

                 )}

                <p className="text-gray-500 mt-1">{project.department} Department • PM: {project.manager}</p>

              </div>

              <div className="mt-4 sm:mt-0 flex items-center space-x-4">

                <span className={`px-3 py-1 text-sm font-bold rounded-full ${

                    project.approval === 'Approved' ? 'bg-green-100 text-green-800' 

                    : project.approval === 'Rejected' ? 'bg-red-100 text-red-800' 

                    : 'bg-yellow-100 text-yellow-800'

                }`}>{project.approval}</span>

                <span className={`flex items-center text-sm font-semibold`}>

                  <div className={`w-3 h-3 rounded-full mr-2 ${getPriorityColor(project.priority)}`}></div>

                  {project.priority} Priority

                </span>

              </div>

          </div>

      </Card>


      <div className="border-b border-gray-200">

        <nav className="flex flex-wrap -mb-px">

          <TabButton text="Summary" active={activeTab === 'summary'} onClick={() => setActiveTab('summary')} />

          <TabButton text="Schedule" active={activeTab === 'schedule'} onClick={() => setActiveTab('schedule')} />

          <TabButton text="Stakeholders" active={activeTab === 'stakeholders'} onClick={() => setActiveTab('stakeholders')} />

          <TabButton text="Risks & Changes" active={activeTab === 'risks'} onClick={() => setActiveTab('risks')} />

          {(userRole === 'Project Manager' || userRole === 'Higher Management') && 

            <TabButton text="Documents" active={activeTab === 'documents'} onClick={() => setActiveTab('documents')} />

          }

        </nav>

      </div>


      <div className="mt-6">

        {tabContent[activeTab]}

        {activeTab === 'documents' && 

            <Card>

                <SectionTitle icon={<Download size={24}/>} text="Project Documents" />

                <p className="text-gray-600 mb-4">Download the original project lifecycle documents.</p>

                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">

                    <a href="#" className="flex items-center p-3 bg-gray-100 hover:bg-blue-100 rounded-lg transition-colors">

                        <FileText className="text-blue-600 mr-3"/>

                        <span className="font-medium text-gray-800">Business_Case.docx</span>

                    </a>

                </div>

            </Card>

        }

      </div>


    </div>

  )

};


export default function App() {

  const [userRole, setUserRole] = useState("Project Department"); // Project Manager, Project Department, Higher Management

  const [selectedProject, setSelectedProject] = useState(null);

  const [projects, setProjects] = useState(initialProjects);


  const handleProjectSelect = (project) => {

    setSelectedProject(project);

  }


  const handleBackToDashboard = () => {

    setSelectedProject(null);

  }


  const handleAddNewProject = (newProjectData) => {

      const newProject = {

          id: projects.length + 1,

          ...newProjectData,

          status: "Initiation",

          approval: "Pending",

          progress: 0,

          health: "On Track",

          spent: 0,

          businessCase: { problem: "TBD", solution: "TBD", benefits: [] },

          projectCharter: { scopeIn: [], scopeOut: [], successCriteria: "TBD" },

          tasks: [],

          risks: [],

          changeRequests: [],

          stakeholders: []

      }

      setProjects(prev => [...prev, newProject]);

  }


  const handleUpdateProject = (updatedProject) => {

      setProjects(prev => prev.map(p => p.id === updatedProject.id ? updatedProject : p));

      setSelectedProject(updatedProject);

  }


  const RoleBasedWrapper = ({ children }) => {

    const getRoleIcon = () => {

        switch(userRole) {

            case 'Project Manager': return <User size={20} className="text-blue-500" />;

            case 'Project Department': return <Building size={20} className="text-purple-500"/>;

            case 'Higher Management': return <Briefcase size={20} className="text-green-500" />;

            default: return null;

        }

    }

    return (

        <div className="bg-gray-100 min-h-screen font-sans text-gray-900">

            <header className="bg-white shadow-sm p-4 sticky top-0 z-10">

                <div className="max-w-7xl mx-auto flex justify-between items-center">

                    <div className="flex items-center">

                        <GanttChartSquare size={32} className="text-blue-600" />

                        <h1 className="text-2xl font-bold ml-2 text-gray-800">Project Lifecycle Dashboard</h1>

                    </div>

                    <div className="flex items-center space-x-4">

                       <div className="flex items-center space-x-2 bg-gray-100 p-2 rounded-lg">

                           {getRoleIcon()}

                            <select value={userRole} onChange={e => setUserRole(e.target.value)} className="bg-transparent font-semibold focus:outline-none">

                                <option>Project Manager</option>

                                <option>Project Department</option>

                                <option>Higher Management</option>

                            </select>

                       </div>

                    </div>

                </div>

            </header>

            <main className="p-4 sm:p-6 max-w-7xl mx-auto">

                {children}

            </main>

        </div>

    );

  }


  // Filter projects based on role

  let visibleProjects = projects;

  if(userRole === 'Project Manager') {

      visibleProjects = projects.filter(p => ["Alex Green", "Diana Prince"].includes(p.manager));

  } else if (userRole === 'Higher Management') {

      visibleProjects = projects.filter(p => p.approval === 'Pending' || p.health === 'At Risk');

  }


  if (selectedProject) {

    if(userRole === 'Project Manager' && !["Alex Green", "Diana Prince"].includes(selectedProject.manager)){

        return (

            <RoleBasedWrapper>

                 <div className="text-center p-10">

                    <AlertTriangle className="mx-auto text-red-500" size={48} />

                    <h2 className="mt-4 text-2xl font-bold">Access Denied</h2>

                    <p className="mt-2 text-gray-600">You do not have permission to view this project.</p>

                    <button onClick={handleBackToDashboard} className="mt-6 bg-blue-600 text-white px-6 py-2 rounded-lg hover:bg-blue-700">Go to Dashboard</button>

                </div>

            </RoleBasedWrapper>

        )

    }

    return (

        <RoleBasedWrapper>

            <ProjectDetails project={selectedProject} onBack={handleBackToDashboard} userRole={userRole} onUpdateProject={handleUpdateProject}/>

        </RoleBasedWrapper>

    );

  }


  return (

    <RoleBasedWrapper>

        {userRole === 'Higher Management' && (

            <div className="mb-6 p-4 bg-yellow-50 border border-yellow-200 rounded-lg">

                <h3 className="text-lg font-bold text-yellow-800 flex items-center"><AlertTriangle className="mr-2"/>Management View</h3>

                <p className="text-yellow-700">Projects requiring approval or at risk are shown below. Select a project to review and approve/reject.</p>

            </div>

        )}

        <DashboardOverview 

            projects={userRole === 'Project Department' ? projects : visibleProjects} 

            onProjectSelect={handleProjectSelect}

            userRole={userRole}

            onAddNewProject={handleAddNewProject}

        />

    </RoleBasedWrapper>

  );

}
